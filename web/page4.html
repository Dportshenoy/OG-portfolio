Python Assistant

import os
import webbrowser
import pyttsx3

ai = pyttsx3.init('sapi5')

def speech(audio):
ai.say(audio)
ai.runAndWait()

speech("Hello, please let me recognize you")
password = input("Password Please: ")
if 'PYTTSX#' in password:
speech("Welcome Boss. What would you like me to do?")
else:
speech("You are not my Boss! Get out from here you loser!!!")
exit()

while True:
Command = input("Command by the Boss: ")

#opens stackoverflow
elif 'open stackoverflow' in Command:
stack = 'http://www.stackoverflow.com'
speech("Opening...")
os.startfile(stack)

#opens github
elif 'open git' in Command:
git = 'http://www.github.com'
speech("Opening...")
os.startfile(git)

#opens youtube
elif 'Youtube' in Command:
y = 'http://www.youtube.com'
speech("Opening...")
os.startfile(y)

#opens google
elif 'google' in Command:
g = 'http//www.google.com'
speech("Opening...")
os.startfile(g)

#opens Python playlist in youtube
elif 'open pyt' in Command:
webbrowser.open("https://www.youtube.com/playlist?list=PLu0W_9lII9agICnT8t4iYVSZ3eykIAOME")
speech("Opening...")

#Opens Machine Learning Youtube playlist
elif 'open Mac' in Command:
webbrowser.open("https://www.youtube.com/playlist?list=PLu0W_9lII9ai6fAMHp-acBmJONT7Y4BSG")
speech("Opening...")

#Calling JARVIS
elif 'jarvis' in Command:
speech("What is it that I can do for you Boss")

#Appreciation
elif 'good jarvis' in Command:
speech("It is my job to keep you happy boss!")

#opens gmail
elif 'open gmail' in Command:
webbrowser.open("gmail.com")
speech("Opening...")


#Exits program
elif 'xxx' in Command:
exit()

else:
print("Invalid command Boss")
